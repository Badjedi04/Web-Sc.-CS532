$$.handlePlayerApi = function(api) {
	var body = document.body,
		THEATER_MODE_CLASS = 'theater-mode',
		resizeMode = api.getResizeMode(),
		p = $$.query('#vid_' + api.getCurrentVideo().id)[0],
		d=$$.query('#videoPlayer')[0],
		dt = $$.query('#theaterVideoPlayer')[0],
		ps = $$.query('#player-spacer')[0],
		MIN_THEATER_MODE_WIDTH = 900,
		theaterModeEnabled = window.innerWidth >= MIN_THEATER_MODE_WIDTH;


	var is_theater_mode = false;

	function resizePlayerSpacer() {
		if(Math.abs(ps.offsetHeight - p.offsetHeight) > 1) {
			ps.style.height = p.offsetHeight + "px";
		}
	}

	resizePlayerSpacer();

	function resizeTheaterMode() {
		if( is_theater_mode ) {
			dt.style.maxHeight = Math.floor(
				Math.min(
					Math.max(window.innerHeight - 56, 80),
					dt.clientWidth * 9 / 16
				)
			) + "px";
		}
	}

	api.on("theaterMode", function(do_theater_mode) {
		is_theater_mode = do_theater_mode;

		// NOTE: This will fire regardless of the current page width!

		if ( do_theater_mode ) {
			body.classList.add(THEATER_MODE_CLASS);
			dt.appendChild(p);
			api.setResizeMode('full');
			resizeTheaterMode();
		} else {
			body.classList.remove(THEATER_MODE_CLASS);
			d.appendChild(p);
			api.setResizeMode(resizeMode);
		}
	});

	// Tell player to show Theater Mode button
	if (theaterModeEnabled) {api.sendEvent('enableTheaterMode')}

	api.on('resize', function() {
		resizePlayerSpacer();
		resizeTheaterMode();
	});

	window.addEventListener('resize', function (event) {
		if (theaterModeEnabled && window.innerWidth < MIN_THEATER_MODE_WIDTH) {
			api.sendEvent('disableTheaterMode');
			theaterModeEnabled = false;
		} else if (!theaterModeEnabled && window.innerWidth >= MIN_THEATER_MODE_WIDTH) {
			api.sendEvent('enableTheaterMode');
			theaterModeEnabled = true;
		}
	});
}
